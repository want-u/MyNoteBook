```

# @File    : 反转文档.py

file = 'passwords.txt'


def get_content():
    with open(file, 'rb') as f:
        a = str(f.read())[1:-1]
        b = a.split('\\r\\n')
        a = b if len(b) > 1 else a.split('\\n')
        a.reverse()
        for i in a:
            yield i + '\n'


def new_write():
    with open(f'reverse_{file}', 'w') as f2:
        for i in get_content():
            f2.write(i)
        print('反转完成...')


if __name__ == '__main__':
    new_write()
```
