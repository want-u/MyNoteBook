### Liveness探测

```
[root@k8s-master ~]# cat livenesspod.yml 
apiVersion: v1
kind: Pod
metadata:
  labels:
    run: healthcheck
  name: healthcheck
spec:
  restartPolicy: OnFailure
  containers:
  - image: busybox
    name: healthcheck
    args: ["/bin/sh","-c","touch /tmp/healty; sleep 30; rm -f /tmp/healty; sleep 99999"]
    livenessProbe:
      exec:
        command:
        - cat
        - /tmp/healty
      initialDelaySeconds: 10
      periodSeconds: 5

# 应用yaml文件
[root@k8s-master ~]# kubectl apply -f livenesspod.yml
# 观察容器状态[重启次数]
[root@k8s-master ~]# watch kubectl get pod -o wide
# 观察事件过程
[root@k8s-master ~]# kubectl describe pod healthcheck
# 在线修改参数配置
[root@k8s-master ~]# kubectl edit pod healthcheck
# 删除pod
[root@k8s-master ~]# kubectl delete pod healthcheck
```

### 在Scale Up 中使用Health Check

```
[root@k8s-master ~]# kubectl create deployment web --image=httpd --dry-run="client" -o yaml > web.yml
[root@k8s-master ~]# vim web.yml 
[root@k8s-master ~]# cat web.yml 
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: web
  name: web
spec:
  replicas: 3
  selector:
    matchLabels:
      app: web
  template:
    metadata:
      labels:
        app: web
    spec:
      containers:
      - image: httpd
        name: httpd
        ports:
        -containerPort: 80
        readinessProbe:
          httpGet:
            scheme: HTTP
            path: /healthy
            port: 80
          initialDelaySeconds: 10
          periodSeconds: 5

```
